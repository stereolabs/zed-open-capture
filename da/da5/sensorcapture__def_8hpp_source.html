181inline std::string wstr2str( const wchar_t* wstr)
182{
183 if(wstr==NULL)
184 return std::string();
185
186 try
187 {
188 std::wstring ws( wstr );
189 std::string str( ws.begin(), ws.end() );
190 return str;
191 }
192 catch(...)
193 {
194 std::cerr << "Failure reading USB data. Please install the udev rules available in the `udev` folder" << std::endl;
195 exit(-1);
196 }
197
198 return std::string();
199}
