1397  SI_CHAR *& a_pData,
1398  const SI_CHAR *& a_pSection,
1399  const SI_CHAR *& a_pKey,
1400  const SI_CHAR *& a_pVal,
1401  const SI_CHAR *& a_pComment
1402  ) const {
1403  a_pComment = NULL;
1404 
1405  SI_CHAR * pTrail = NULL;
1406  while (*a_pData) {
1407  // skip spaces and empty lines
1408  while (*a_pData && IsSpace(*a_pData)) {
1409  ++a_pData;
1410  }
1411  if (!*a_pData) {
1412  break;
1413  }
1414 
1415  // skip processing of comment lines but keep a pointer to
1416  // the start of the comment.
1417  if (IsComment(*a_pData)) {
1418  LoadMultiLineText(a_pData, a_pComment, NULL, true);
1419  continue;
1420  }
1421 
1422  // process section names
1423  if (*a_pData == '[') {
1424  // skip leading spaces
1425  ++a_pData;
1426  while (*a_pData && IsSpace(*a_pData)) {
1427  ++a_pData;
1428  }
1429 
1430  // find the end of the section name (it may contain spaces)
1431  // and convert it to lowercase as necessary
1432  a_pSection = a_pData;
1433  while (*a_pData && *a_pData != ']' && !IsNewLineChar(*a_pData)) {
1434  ++a_pData;
1435  }
1436 
1437  // if it's an invalid line, just skip it
1438  if (*a_pData != ']') {
1439  continue;
1440  }
1441 
1442  // remove trailing spaces from the section
1443  pTrail = a_pData - 1;
1444  while (pTrail >= a_pSection && IsSpace(*pTrail)) {
1445  --pTrail;
1446  }
1447  ++pTrail;
1448  *pTrail = 0;
1449 
1450  // skip to the end of the line
1451  ++a_pData; // safe as checked that it == ']' above
1452  while (*a_pData && !IsNewLineChar(*a_pData)) {
1453  ++a_pData;
1454  }
1455 
1456  a_pKey = NULL;
1457  a_pVal = NULL;
1458  return true;
1459  }
1460 
1461  // find the end of the key name (it may contain spaces)
1462  // and convert it to lowercase as necessary
1463  a_pKey = a_pData;
1464  while (*a_pData && *a_pData != '=' && !IsNewLineChar(*a_pData)) {
1465  ++a_pData;
1466  }
1467 
1468  // if it's an invalid line, just skip it
1469  if (*a_pData != '=') {
1470  continue;
1471  }
1472 
1473  // empty keys are invalid
1474  if (a_pKey == a_pData) {
1475  while (*a_pData && !IsNewLineChar(*a_pData)) {
1476  ++a_pData;
1477  }
1478  continue;
1479  }
1480 
1481  // remove trailing spaces from the key
1482  pTrail = a_pData - 1;
1483  while (pTrail >= a_pKey && IsSpace(*pTrail)) {
1484  --pTrail;
1485  }
1486  ++pTrail;
1487  *pTrail = 0;
1488 
1489  // skip leading whitespace on the value
1490  ++a_pData; // safe as checked that it == '=' above
1491  while (*a_pData && !IsNewLineChar(*a_pData) && IsSpace(*a_pData)) {
1492  ++a_pData;
1493  }
1494 
1495  // find the end of the value which is the end of this line
1496  a_pVal = a_pData;
1497  while (*a_pData && !IsNewLineChar(*a_pData)) {
1498  ++a_pData;
1499  }
1500 
1501  // remove trailing spaces from the value
1502  pTrail = a_pData - 1;
1503  if (*a_pData) { // prepare for the next round
1504  SkipNewLine(a_pData);
1505  }
1506  while (pTrail >= a_pVal && IsSpace(*pTrail)) {
1507  --pTrail;
1508  }
1509  ++pTrail;
1510  *pTrail = 0;
1511 
1512  // check for multi-line entries
1513  if (m_bAllowMultiLine && IsMultiLineTag(a_pVal)) {
1514  // skip the "<<<" to get the tag that will end the multiline
1515  const SI_CHAR * pTagName = a_pVal + 3;
1516  return LoadMultiLineText(a_pData, a_pVal, pTagName);
1517  }
1518 
1519  // return the standard entry
1520  return true;
1521  }
1522 
1523  return false;
1524 }

1582 template<class SI_CHAR, class SI_STRLESS, class SI_CONVERTER>
1583 bool
1584 CSimpleIniTempl<SI_CHAR, SI_STRLESS, SI_CONVERTER>::LoadMultiLineText(
1585  SI_CHAR *& a_pData,
1586  const SI_CHAR *& a_pVal,
1587  const SI_CHAR * a_pTagName,
1588  bool a_bAllowBlankLinesInComment
1589  ) const {
1590  // we modify this data to strip all newlines down to a single '\n'
1591  // character. This means that on Windows we need to strip out some
1592  // characters which will make the data shorter.
1593  // i.e. LINE1-LINE1\r\nLINE2-LINE2\0 will become
1594  // LINE1-LINE1\nLINE2-LINE2\0
1595  // The pDataLine entry is the pointer to the location in memory that
1596  // the current line needs to start to run following the existing one.
1597  // This may be the same as pCurrLine in which case no move is needed.
1598  SI_CHAR * pDataLine = a_pData;
1599  SI_CHAR * pCurrLine;
1600 
1601  // value starts at the current line
1602  a_pVal = a_pData;
1603 
1604  // find the end tag. This tag must start in column 1 and be
1605  // followed by a newline. No whitespace removal is done while
1606  // searching for this tag.
1607  SI_CHAR cEndOfLineChar = *a_pData;
1608  for (;;) {
1609  // if we are loading comments then we need a comment character as
1610  // the first character on every line
1611  if (!a_pTagName && !IsComment(*a_pData)) {
1612  // if we aren't allowing blank lines then we're done
1613  if (!a_bAllowBlankLinesInComment) {
1614  break;
1615  }
1616 
1617  // if we are allowing blank lines then we only include them
1618  // in this comment if another comment follows, so read ahead
1619  // to find out.
1620  SI_CHAR * pCurr = a_pData;
1621  int nNewLines = 0;
1622  while (IsSpace(*pCurr)) {
1623  if (IsNewLineChar(*pCurr)) {
1624  ++nNewLines;
1625  SkipNewLine(pCurr);
1626  } else {
1627  ++pCurr;
1628  }
1629  }
1630 
1631  // we have a comment, add the blank lines to the output
1632  // and continue processing from here
1633  if (IsComment(*pCurr)) {
1634  for (; nNewLines > 0; --nNewLines) *pDataLine++ = '\n';
1635  a_pData = pCurr;
1636  continue;
1637  }
1638 
1639  // the comment ends here
1640  break;
1641  }
1642 
1643  // find the end of this line
1644  pCurrLine = a_pData;
1645  while (*a_pData && !IsNewLineChar(*a_pData)) ++a_pData;
1646 
1647  // move this line down to the location that it should be if necessary
1648  if (pDataLine < pCurrLine) {
1649  size_t nLen = (size_t) (a_pData - pCurrLine);
1650  memmove(pDataLine, pCurrLine, nLen * sizeof (SI_CHAR));
1651  pDataLine[nLen] = '\0';
1652  }
1653 
1654  // end the line with a NULL
1655  cEndOfLineChar = *a_pData;
1656  *a_pData = 0;
1657 
1658  // if are looking for a tag then do the check now. This is done before
1659  // checking for end of the data, so that if we have the tag at the end
1660  // of the data then the tag is removed correctly.
1661  if (a_pTagName &&
1662  (!IsLess(pDataLine, a_pTagName) && !IsLess(a_pTagName, pDataLine))) {
1663  break;
1664  }
1665 
1666  // if we are at the end of the data then we just automatically end
1667  // this entry and return the current data.
1668  if (!cEndOfLineChar) {
1669  return true;
1670  }
1671 
1672  // otherwise we need to process this newline to ensure that it consists
1673  // of just a single \n character.
1674  pDataLine += (a_pData - pCurrLine);
1675  *a_pData = cEndOfLineChar;
1676  SkipNewLine(a_pData);
1677  *pDataLine++ = '\n';
1678  }
1679 
1680  // if we didn't find a comment at all then return false
1681  if (a_pVal == a_pData) {
1682  a_pVal = NULL;
1683  return false;
1684  }
1685 
1686  // the data (which ends at the end of the last line) needs to be
1687  // null-terminated BEFORE before the newline character(s). If the
1688  // user wants a new line in the multi-line data then they need to
1689  // add an empty line before the tag.
1690  *--pDataLine = '\0';
1691 
1692  // if looking for a tag and if we aren't at the end of the data,
1693  // then move a_pData to the start of the next line.
1694  if (a_pTagName && cEndOfLineChar) {
1695  SI_ASSERT(IsNewLineChar(cEndOfLineChar));
1696  *a_pData = cEndOfLineChar;
1697  SkipNewLine(a_pData);
1698  }
1699 
1700  return true;
1701 }
